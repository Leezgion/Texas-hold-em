## 3️⃣ 断线重连处理实施

### 步骤 1: 创建 ReconnectingOverlay 组件

已创建文件：`client/src/components/ReconnectingOverlay.jsx`

### 步骤 2: 在 GameContext 中添加重连状态

修改 `GameContext.jsx` 的 store：

```js
const useGameStore = create((set, get) => ({
  // ... 现有状态
  
  // 新增状态
  isReconnecting: false,
  reconnectAttempts: 0,

  connectSocket: () => {
    const socket = io(serverUrl, {
      reconnection: true,
      reconnectionDelay: 1000,
      reconnectionDelayMax: 5000,
      reconnectionAttempts: Infinity,
    });

    // 监听重连事件
    socket.io.on('reconnect_attempt', (attemptNumber) => {
      set({ isReconnecting: true, reconnectAttempts: attemptNumber });
    });

    socket.io.on('reconnect', () => {
      set({ isReconnecting: false, reconnectAttempts: 0 });
      
      // 重新加入房间
      const { roomId, deviceId } = get();
      if (roomId) {
        socket.emit('rejoinRoom', { roomId, deviceId });
      }
    });

    socket.on('disconnect', () => {
      set({ connected: false, isReconnecting: true });
    });

    return socket;
  },

  // 手动重连
  manualReconnect: () => {
    const { socket } = get();
    if (socket && !socket.connected) {
      socket.connect();
    }
  },
}));
```

### 步骤 3: 在 App 中使用 ReconnectingOverlay


### 步骤 4: 服务器端支持

在服务器添加 `rejoinRoom` 事件处理：

```js
// server/server.js
socket.on('rejoinRoom', async ({ roomId, deviceId }) => {
  const room = roomManager.getRoom(roomId);
  if (room) {
    // 恢复玩家连接
    const player = room.players.find(p => p.deviceId === deviceId);
    if (player) {
      player.socketId = socket.id;
      socket.join(roomId);
      // 发送最新游戏状态
      socket.emit('gameStateUpdate', room.getState());
    }
  }
});
```

## 4️⃣ 键盘快捷键实施

### 步骤 1: 创建 useKeyboardShortcuts Hook

创建文件 `client/src/hooks/useKeyboardShortcuts.js`：

```js
import { useEffect } from 'react';

export const useKeyboardShortcuts = (canAct, handlers) => {
  useEffect(() => {
    if (!canAct) return;

    const handleKeyPress = (e) => {
      // 忽略输入框中的按键
      if (e.target.tagName === 'INPUT' || e.target.tagName === 'TEXTAREA') {
        return;
      }

      const key = e.key.toLowerCase();

      switch (key) {
        case 'f':
          handlers.onFold?.();
          break;
        case 'c':
          handlers.onCheckCall?.();
          break;
        case 'r':
          handlers.onRaise?.();
          break;
        case 'a':
          handlers.onAllIn?.();
          break;
        default:
          break;
      }
    };

    window.addEventListener('keydown', handleKeyPress);
    return () => window.removeEventListener('keydown', handleKeyPress);
  }, [canAct, handlers]);
};
```

### 步骤 2: 在 ActionButtons 中使用

修改 `ActionButtons.jsx`：

## 6️⃣ 获胜牌高亮实施

### 步骤 1: 在 HandResultModal 中添加高亮逻辑

修改 `HandResultModal.jsx`：

```jsx
const HandResultModal = () => {
  const { handResult } = useGame();

  // 获取获胜牌的索引
  const getWinningCardIndices = (hand) => {
    // 假设服务器返回 winningCards 数组
    return hand.winningCards || [];
  };

  return (
    <Modal {...props}>
      <div className="space-y-6">
        {handResult.hands?.map((hand, index) => (
          <div key={index} className="bg-gray-700 p-4 rounded-lg">
            <div className="text-lg font-semibold mb-3">
              {hand.playerName} - {hand.handName}
            </div>
            
            {/* 玩家手牌 */}
            <div className="flex space-x-2 mb-3">
              {hand.cards.map((card, cardIndex) => {
                const isWinning = getWinningCardIndices(hand).includes(cardIndex);
                return (
                  <Card
                    key={cardIndex}
                    card={card}
                    className={isWinning ? 'winning-card' : ''}
                  />
                );
              })}
            </div>

            {/* 公共牌 */}
            <div className="flex space-x-2">
              {handResult.communityCards.map((card, cardIndex) => {
                const isWinning = getWinningCardIndices(hand).includes(cardIndex + hand.cards.length);
                return (
                  <Card
                    key={cardIndex}
                    card={card}
                    size="small"
                    className={isWinning ? 'winning-card' : ''}
                  />
                );
              })}
            </div>
          </div>
        ))}
      </div>
    </Modal>
  );
};
```

## 💡 常见问题

**Q: Toast 通知不显示？**
A: 检查 ToastContainer 是否已添加到 App.jsx，并确保 useToast hook 正确导入。

**Q: 断线重连后游戏状态丢失？**
A: 确保服务器实现了 rejoinRoom 事件，并正确恢复玩家状态。

**Q: 键盘快捷键在输入框中也触发？**
A: 在事件处理器中添加检查：`if (e.target.tagName === 'INPUT') return;`

**Q: 移动端键盘快捷键怎么办？**
A: 移动端没有物理键盘，快捷键主要是桌面端的优化，移动端继续使用触摸操作。
