### 2. GameContext.jsx - 重连逻辑

```jsx
const useGameStore = create((set, get) => ({
  // 新增状态
  isReconnecting: false,
  reconnectAttempts: 0,

  connectSocket: () => {
    const socket = io(serverUrl, {
      reconnection: true,
      reconnectionDelay: 1000,
      reconnectionAttempts: Infinity,
    });

    // 重连监听
    socket.io.on('reconnect_attempt', (num) => {
      set({ isReconnecting: true, reconnectAttempts: num });
    });

    socket.io.on('reconnect', () => {
      set({ isReconnecting: false, reconnectAttempts: 0 });
      const { roomId, deviceId } = get();
      if (roomId) socket.emit('rejoinRoom', { roomId, deviceId });
    });

    // 错误处理改为事件
    socket.on('error', (msg) => {
      window.dispatchEvent(new CustomEvent('game-error', { detail: msg }));
    });
  },
}));
```

### 3. GameRoom.jsx - 游戏日志

```jsx
import GameLog from './GameLog';

const GameRoom = () => {
  const [gameLogs, setGameLogs] = useState([]);

  useEffect(() => {
    if (gameState?.lastAction) {
      const player = players.find(p => p.id === gameState.lastAction.playerId);
      if (player) {
        setGameLogs(prev => [...prev, {
          player: player.nickname,
          action: gameState.lastAction.action,
          amount: gameState.lastAction.amount,
          timestamp: Date.now(),
        }]);
      }
    }
  }, [gameState?.lastAction]);

  return (
    <div>
      {/* 在侧边栏添加 */}
      <GameLog logs={gameLogs} />
    </div>
  );
};
```

## 📡 Socket.io 事件新增

### 客户端发送

```js

```

### 服务器端处理（需添加）

```js
socket.on('rejoinRoom', ({ roomId, deviceId }) => {
  const room = roomManager.getRoom(roomId);
  if (room) {
    const player = room.players.find(p => p.deviceId === deviceId);
    if (player) {
      player.socketId = socket.id;
      socket.join(roomId);
      socket.emit('gameStateUpdate', room.getState());
    }
  }
});
```

## 🐛 常见问题快速解决

### Toast 不显示

- 检查 `ToastContainer` 是否在 `App.jsx` 中
- 确认 `useToast` hook 正确导入
- 查看浏览器控制台是否有错误

### 断线重连失败

- 检查 Socket.io 配置中 `reconnection: true`
- 确认服务器实现了 `rejoinRoom` 事件
- 检查网络连接和服务器状态

### 键盘快捷键不响应

- 确认 `canAct` 为 true
- 检查输入框检测逻辑
- 查看事件监听器是否正确添加

### 游戏日志不更新

- 确认 `gameState.lastAction` 存在
- 检查 `useEffect` 依赖项
- 验证服务器发送的数据结构
