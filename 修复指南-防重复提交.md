# 🔧 快速修复指南：防重复提交机制

## 问题描述

当前用户可以在服务器响应前快速连续点击操作按钮（如"加注"），导致：

- 发送多个重复的 `playerAction` 请求
- 服务器拒绝第二个请求（"不是你的回合"）
- 用户看到混乱的错误提示

## 解决方案

在客户端添加提交锁机制，发送操作后立即禁用UI，直到收到服务器响应。

## 完整的修改后代码

```javascript
import React, { useState, useEffect } from 'react';
import { useGame } from '../contexts/GameContext';
import SliderInput from './SliderInput';

const ActionButtons = ({ player, gameState, currentPlayerId, players }) => {
  const { playerAction } = useGame();
  const [raiseAmount, setRaiseAmount] = useState('');
  const [showRaiseInput, setShowRaiseInput] = useState(false);
  const [sliderValue, setSliderValue] = useState(0);
  const [isSubmitting, setIsSubmitting] = useState(false); // 🆕 提交锁

  // 检查是否是当前玩家的回合
  const isCurrentTurn = gameState && 
    gameState.currentPlayerIndex !== undefined && 
    gameState.currentPlayerIndex === players.findIndex((p) => p.id === currentPlayerId);

  // 🔧 添加 isSubmitting 检查
  const canAct = isCurrentTurn && !player.folded && !player.allIn && !isSubmitting;

  const canCheck = canAct && player.currentBet >= gameState.currentBet;
  const canRaise = canAct && player.chips > gameState.currentBet;
  const callAmount = gameState.currentBet - player.currentBet;

  // 🆕 收到新状态时解锁
  useEffect(() => {
    if (gameState) {
      setIsSubmitting(false);
    }
  }, [gameState]);

  // 🆕 超时保护
  useEffect(() => {
    if (isSubmitting) {
      const timeout = setTimeout(() => {
        console.warn('操作超时，自动解锁UI');
        setIsSubmitting(false);
      }, 5000);

      return () => clearTimeout(timeout);
    }
  }, [isSubmitting]);

  const handleAction = (action, amount = 0) => {
    if (canAct && !isSubmitting) {  // 🔧 双重检查
      setIsSubmitting(true);  // 🔧 立即锁定
      playerAction(action, amount);
      setShowRaiseInput(false);
      setRaiseAmount('');
      setSliderValue(gameState?.minRaise || 0);
    }
  };

  const handleSliderChange = (value) => {
    setSliderValue(value);
    setRaiseAmount(value.toString());
  };

  const handleRaiseInputChange = (e) => {
    const value = e.target.value;
    setRaiseAmount(value);
    if (value && !isNaN(value)) {
      setSliderValue(parseInt(value));
    }
  };

  const handleRaise = () => {
    const amount = parseInt(raiseAmount);
    if (amount >= gameState.minRaise && amount <= player.chips) {
      handleAction('raise', amount);
    }
  };

  const handleAllIn = () => {
    handleAction('raise', player.chips);
  };

  if (!isCurrentTurn || player.folded || player.allIn) {
    return null;
  }

  return (
    <div className="flex flex-col items-center space-y-3 max-w-sm mx-auto">
      {/* 🆕 提交状态提示 */}
      {isSubmitting && (
        <div className="text-xs text-yellow-400 animate-pulse">
          处理中...
        </div>
      )}

      {/* 操作按钮 */}
      <div className="flex items-center gap-2">
        <button
          onClick={() => handleAction('fold')}
          disabled={!canAct}
          className={`w-11 h-11 bg-red-600/90 text-white rounded-xl font-bold transition-all ${
            !canAct ? 'opacity-50 cursor-not-allowed' : 'hover:bg-red-500 hover:scale-105'
          }`}
          title="弃牌 (F)"
        >
          弃
        </button>

        {canCheck ? (
          <button
            onClick={() => handleAction('check')}
            disabled={!canAct}
            className={`h-11 px-4 bg-green-600/90 text-white rounded-xl font-bold transition-all ${
              !canAct ? 'opacity-50 cursor-not-allowed' : 'hover:bg-green-500 hover:scale-105'
            }`}
            title="过牌 (C)"
          >
            过牌
          </button>
        ) : (
          <button
            onClick={() => handleAction('call')}
            disabled={!canAct}
            className={`h-11 px-4 bg-blue-600/90 text-white rounded-xl font-bold transition-all ${
              !canAct ? 'opacity-50 cursor-not-allowed' : 'hover:bg-blue-500 hover:scale-105'
            }`}
            title="跟注 (C)"
          >
            跟 {callAmount}
          </button>
        )}

        {canRaise && !showRaiseInput && (
          <button
            onClick={() => setShowRaiseInput(true)}
            disabled={!canAct}
            className={`h-11 px-4 bg-yellow-600/90 text-white rounded-xl font-bold transition-all ${
              !canAct ? 'opacity-50 cursor-not-allowed' : 'hover:bg-yellow-500 hover:scale-105'
            }`}
            title="加注 (R)"
          >
            加注
          </button>
        )}

        {canRaise && (
          <button
            onClick={handleAllIn}
            disabled={!canAct}
            className={`h-11 px-4 bg-purple-600/90 text-white rounded-xl font-bold transition-all ${
              !canAct ? 'opacity-50 cursor-not-allowed' : 'hover:bg-purple-500 hover:scale-105'
            }`}
            title="All-in"
          >
            全押 {player.chips}
          </button>
        )}
      </div>

      {/* 加注输入区域 */}
      {showRaiseInput && canRaise && (
        <div className="w-full bg-black bg-opacity-50 rounded-lg p-3 border border-yellow-600">
          <div className="flex items-center justify-between mb-2">
            <label className="text-white text-sm">加注金额:</label>
            <button
              onClick={() => {
                setShowRaiseInput(false);
                setRaiseAmount('');
              }}
              className="text-gray-400 hover:text-white"
            >
              ✕
            </button>
          </div>

          <SliderInput
            min={gameState.minRaise}
            max={player.chips}
            value={sliderValue}
            onChange={handleSliderChange}
          />

          <div className="flex items-center gap-2 mt-3">
            <input
              type="number"
              value={raiseAmount}
              onChange={handleRaiseInputChange}
              min={gameState.minRaise}
              max={player.chips}
              className="flex-1 bg-gray-800 text-white px-3 py-2 rounded-lg"
              placeholder={`最小 ${gameState.minRaise}`}
            />
            <button
              onClick={handleRaise}
              disabled={!canAct || !raiseAmount || parseInt(raiseAmount) < gameState.minRaise}
              className={`px-4 py-2 bg-yellow-600 text-white rounded-lg ${
                !canAct || !raiseAmount || parseInt(raiseAmount) < gameState.minRaise
                  ? 'opacity-50 cursor-not-allowed'
                  : 'hover:bg-yellow-500'
              }`}
            >
              确认
            </button>
          </div>
        </div>
      )}
    </div>
  );
};

export default ActionButtons;
```

### 2. 添加键盘快捷键

```javascript
useEffect(() => {
  const handleKeyPress = (e) => {
    if (!canAct || isSubmitting) return;
    
    switch (e.key.toLowerCase()) {
      case 'f': handleAction('fold'); break;
      case 'c': canCheck ? handleAction('check') : handleAction('call'); break;
      case 'r': setShowRaiseInput(true); break;
    }
  };

  window.addEventListener('keypress', handleKeyPress);
  return () => window.removeEventListener('keypress', handleKeyPress);
}, [canAct, isSubmitting, canCheck]);
```

### 3. 添加操作确认

对于重要操作（如弃牌、All-in），添加二次确认：

```javascript
const [showConfirm, setShowConfirm] = useState(null);

const handleActionWithConfirm = (action, amount = 0) => {
  if (action === 'fold' || amount === player.chips) {
    setShowConfirm({ action, amount });
  } else {
    handleAction(action, amount);
  }
};

const confirmAction = () => {
  if (showConfirm) {
    handleAction(showConfirm.action, showConfirm.amount);
    setShowConfirm(null);
  }
};
```
